实时语音识别与翻译功能实施方案
1. 项目概述
本方案旨在为 Electron + Vue3 翻译软件集成阿里云 DashScope 的 Gummy 实时语音识别与翻译服务。通过 WebSocket 实现全双工通信，支持实时语音输入、即时识别和翻译结果展示。
2. 技术选型
2.1 架构方案：方案A（主进程处理）
选择理由：
稳定性：WebSocket 连接和音频处理在主进程中运行，不受渲染进程页面刷新/崩溃影响
性能：直接使用 Node.js Native 模块处理音频流，效率更高
维护性：前后端逻辑分离，符合 Electron 最佳实践
包大小：增加依赖极小（ws库仅几十KB），对最终包体影响可忽略
2.2 技术栈
主进程：Node.js + TypeScript + WebSocket (ws库)
渲染进程：Vue 3 + Composition API + TypeScript
进程通信：Electron IPC (ipcMain/ipcRenderer)
密钥管理：dotenv + 环境变量
音频处理：可选 node-microphone 或 electron 内置 API

// WebSocket 消息类型
export interface WSMessageHeader {
  action?: 'run-task' | 'finish-task';
  event?: 'task-started' | 'result-generated' | 'task-finished' | 'task-failed';
  task_id: string;
  streaming: 'duplex';
  error_code?: string;
  error_message?: string;
  attributes?: Record<string, any>;
}

export interface WordTimestamp {
  begin_time: number;
  end_time: number;
  text: string;
  punctuation: string;
  fixed: boolean;
  speaker_id: number | null;
}

export interface TranscriptionResult {
  sentence_id: number;
  begin_time: number;
  end_time: number;
  text: string;
  words: WordTimestamp[];
  sentence_end: boolean;
}

export interface TranslationResult extends TranscriptionResult {
  lang: string;
  pre_end_failed?: boolean;
}

export interface ASROutput {
  transcription?: TranscriptionResult;
  translations?: TranslationResult[];
}

export interface WSMessagePayload {
  task_group?: 'audio';
  task?: 'asr';
  function?: 'recognition';
  model?: string;
  parameters?: {
    sample_rate: number;
    format: string;
    source_language?: string;
    transcription_enabled: boolean;
    translation_enabled: boolean;
    translation_target_languages: string[];
    vocabulary_id?: string;
    max_end_silence?: number;
  };
  input?: Record<string, any>;
  output?: ASROutput;
}

export interface WSMessage {
  header: WSMessageHeader;
  payload: WSMessagePayload;
}

// IPC 通信类型
export interface IPCChannels {
  'asr:start': { taskId?: string; targetLanguage?: string };
  'asr:stop': void;
  'asr:result': { transcription?: string; translation?: string; isFinal: boolean };
  'asr:status': { status: 'idle' | 'listening' | 'processing' | 'error'; error?: string };
  'asr:transcription': TranscriptionResult;
  'asr:translation': TranslationResult;
}
4.2 主进程：WebSocket 服务 (src/main/services/ASRService.ts)
import WebSocket from 'ws';
import { v4 as uuidv4 } from 'uuid';
import { EventEmitter } from 'events';
import { config } from '../env';
import { 
  WSMessage, 
  ASROutput, 
  TranscriptionResult, 
  TranslationResult 
} from '../../types/asr';

export class ASRService extends EventEmitter {
  private ws: WebSocket | null = null;
  private taskId: string = '';
  private isTaskStarted: boolean = false;
  private audioStream: any = null; // 麦克风流，根据实际实现调整

  constructor() {
    super();
  }

  /**
   * 开始语音识别任务
   */
  async startRecognition(options: {
    targetLanguage?: string;
    sourceLanguage?: string;
  } = {}): Promise<void> {
    try {
      // 生成任务ID
      this.taskId = uuidv4().replace(/-/g, '').slice(0, 32);
      
      // 建立WebSocket连接
      await this.connectWebSocket();
      
      // 发送 run-task 指令
      await this.sendRunTask({
        targetLanguage: options.targetLanguage || config.targetLanguage,
        sourceLanguage: options.sourceLanguage
      });
      
    } catch (error) {
      this.emit('error', error);
      throw error;
    }
  }

  /**
   * 停止语音识别任务
   */
  async stopRecognition(): Promise<void> {
    if (!this.isTaskStarted) {
      return;
    }

    try {
      // 停止音频流
      if (this.audioStream) {
        this.audioStream.stop();
        this.audioStream = null;
      }

      // 发送 finish-task 指令
      await this.sendFinishTask();

      // 等待任务完成或超时关闭
      setTimeout(() => {
        if (this.ws) {
          this.ws.close();
          this.ws = null;
        }
      }, 3000);

    } catch (error) {
      this.emit('error', error);
    }
  }

  /**
   * 建立WebSocket连接
   */
  private connectWebSocket(): Promise<void> {
    return new Promise((resolve, reject) => {
      this.ws = new WebSocket(config.wsUrl, {
        headers: {
          Authorization: `bearer ${config.apiKey}`,
          'User-Agent': 'Electron-Translation-App/1.0'
        }
      });

      this.ws.on('open', () => {
        console.log('WebSocket连接已建立');
        resolve();
      });

      this.ws.on('message', (data: Buffer) => {
        this.handleServerMessage(data.toString());
      });

      this.ws.on('error', (error) => {
        this.emit('error', `WebSocket错误: ${error.message}`);
        reject(error);
      });

      this.ws.on('close', () => {
        this.isTaskStarted = false;
        this.emit('status', { status: 'idle' });
      });
    });
  }

  /**
   * 发送 run-task 指令
   */
  private sendRunTask(options: {
    targetLanguage: string;
    sourceLanguage?: string;
  }): void {
    if (!this.ws) {
      throw new Error('WebSocket连接未建立');
    }

    const runTaskMsg: WSMessage = {
      header: {
        action: 'run-task',
        task_id: this.taskId,
        streaming: 'duplex'
      },
      payload: {
        task_group: 'audio',
        task: 'asr',
        function: 'recognition',
        model: 'gummy-realtime-v1',
        parameters: {
          sample_rate: config.audioSampleRate,
          format: config.audioFormat,
          source_language: options.sourceLanguage || null,
          transcription_enabled: true,
          translation_enabled: true,
          translation_target_languages: [options.targetLanguage],
          max_end_silence: 800
        },
        input: {}
      }
    };

    this.ws.send(JSON.stringify(runTaskMsg));
    console.log('已发送 run-task 指令');
  }

  /**
   * 处理服务器消息
   */
  private handleServerMessage(data: string): void {
    try {
      const message: WSMessage = JSON.parse(data);
      const event = message.header.event;

      switch (event) {
        case 'task-started':
          console.log('任务已开始，准备发送音频流');
          this.isTaskStarted = true;
          this.emit('status', { status: 'listening' });
          this.startAudioStream();
          break;

        case 'result-generated':
          this.handleResultGenerated(message.payload.output);
          break;

        case 'task-finished':
          console.log('任务已完成');
          this.isTaskStarted = false;
          this.emit('status', { status: 'idle' });
          break;

        case 'task-failed':
          console.error('任务失败:', message.header.error_message);
          this.emit('error', message.header.error_message);
          this.emit('status', { 
            status: 'error', 
            error: message.header.error_message 
          });
          break;
      }
    } catch (error) {
      console.error('解析服务器消息失败:', error);
    }
  }

  /**
   * 处理识别/翻译结果
   */
  private handleResultGenerated(output?: ASROutput): void {
    if (!output) return;

    // 发送识别结果
    if (output.transcription) {
      this.emit('transcription', output.transcription);
      
      // 同时发送简化的文本结果用于UI显示
      this.emit('result', {
        transcription: output.transcription.text,
        isFinal: output.transcription.sentence_end
      });
    }

    // 发送翻译结果
    if (output.translations && output.translations.length > 0) {
      const translation = output.translations[0];
      this.emit('translation', translation);
      
      this.emit('result', {
        translation: translation.text,
        isFinal: translation.sentence_end
      });
    }
  }

  /**
   * 开始音频流捕获和发送
   */
  private startAudioStream(): void {
    // 实现取决于选择的音频库
    // 示例：使用 node-microphone
    // this.audioStream = this.createMicrophoneStream();
    console.log('开始音频流捕获（需根据实际音频库实现）');
    
    // 模拟音频发送（实际项目中替换为真实音频流）
    this.simulateAudioStream();
  }

  /**
   * 模拟音频流发送（开发测试用）
   */
  private simulateAudioStream(): void {
    // 注意：实际项目中使用真实麦克风音频流
    // 这里只是示例，展示如何定时发送数据
    const interval = setInterval(() => {
      if (!this.isTaskStarted || !this.ws) {
        clearInterval(interval);
        return;
      }
      
      // 发送静音或测试音频数据
      // 实际实现中，这里应该发送从麦克风捕获的真实音频数据
      const silentData = Buffer.alloc(3200); // 100ms的16kHz 16位单声道静音
      this.ws.send(silentData);
      
    }, 100); // 每100ms发送一次
  }

  /**
   * 发送 finish-task 指令
   */
  private sendFinishTask(): void {
    if (!this.ws || !this.isTaskStarted) {
      return;
    }

    const finishTaskMsg: WSMessage = {
      header: {
        action: 'finish-task',
        task_id: this.taskId,
        streaming: 'duplex'
      },
      payload: {
        input: {}
      }
    };

    this.ws.send(JSON.stringify(finishTaskMsg));
    console.log('已发送 finish-task 指令');
  }

  /**
   * 清理资源
   */
  destroy(): void {
    this.stopRecognition();
    this.removeAllListeners();
  }
}
4.3 主进程：IPC 桥接 (src/main/ipc/asrBridge.ts)
import { ipcMain, BrowserWindow } from 'electron';
import { ASRService } from '../services/ASRService';
import { IPCChannels } from '../../types/asr';

export class ASRIPC {
  private asrService: ASRService;
  private mainWindow: BrowserWindow;

  constructor(mainWindow: BrowserWindow) {
    this.asrService = new ASRService();
    this.mainWindow = mainWindow;
    this.setupIPCHandlers();
    this.setupEventForwarding();
  }

  /**
   * 设置IPC处理器
   */
  private setupIPCHandlers(): void {
    // 开始语音识别
    ipcMain.handle('asr:start', async (_, options: { targetLanguage?: string }) => {
      try {
        await this.asrService.startRecognition(options);
        return { success: true, taskId: this.asrService['taskId'] };
      } catch (error) {
        return { success: false, error: error.message };
      }
    });

    // 停止语音识别
    ipcMain.handle('asr:stop', async () => {
      try {
        await this.asrService.stopRecognition();
        return { success: true };
      } catch (error) {
        return { success: false, error: error.message };
      }
    });

    // 获取识别状态
    ipcMain.handle('asr:getStatus', () => {
      return { status: this.asrService['isTaskStarted'] ? 'listening' : 'idle' };
    });
  }

  /**
   * 设置事件转发
   */
  private setupEventForwarding(): void {
    // 转发识别结果
    this.asrService.on('result', (result) => {
      this.mainWindow.webContents.send('asr:result', result);
    });

    // 转发详细转录结果
    this.asrService.on('transcription', (transcription) => {
      this.mainWindow.webContents.send('asr:transcription', transcription);
    });

    // 转发详细翻译结果
    this.asrService.on('translation', (translation) => {
      this.mainWindow.webContents.send('asr:translation', translation);
    });

    // 转发状态更新
    this.asrService.on('status', (status) => {
      this.mainWindow.webContents.send('asr:status', status);
    });

    // 转发错误
    this.asrService.on('error', (error) => {
      this.mainWindow.webContents.send('asr:error', error);
    });
  }

  /**
   * 清理资源
   */
  destroy(): void {
    this.asrService.destroy();
    ipcMain.removeHandler('asr:start');
    ipcMain.removeHandler('asr:stop');
    ipcMain.removeHandler('asr:getStatus');
  }
}

<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted } from 'vue';
import { IPCChannels } from '../../types/asr';

// 状态管理
const isRecording = ref(false);
const status = ref<'idle' | 'listening' | 'processing' | 'error'>('idle');
const error = ref<string>('');

// 识别结果
const transcriptionText = ref('');
const translationText = ref('');
const isFinalTranscription = ref(false);
const isFinalTranslation = ref(false);
const transcriptionDetails = ref<IPCChannels['asr:transcription'] | null>(null);
const translationDetails = ref<IPCChannels['asr:translation'] | null>(null);

// 配置
const targetLanguage = ref('en');

// 计算属性
const statusText = computed(() => {
  const texts = {
    idle: '准备就绪',
    listening: '录音中...',
    processing: '处理中...',
    error: '发生错误'
  };
  return texts[status.value];
});

const targetLanguageName = computed(() => {
  const names: Record<string, string> = {
    en: '英语',
    ja: '日语',
    ko: '韩语',
    fr: '法语',
    de: '德语'
  };
  return names[targetLanguage.value] || targetLanguage.value;
});

// 方法
const toggleRecording = async () => {
  if (isRecording.value) {
    await stopRecording();
  } else {
    await startRecording();
  }
};

const startRecording = async () => {
  try {
    resetResults();
    error.value = '';
    status.value = 'listening';
    
    const response = await window.ipcRenderer.invoke('asr:start', {
      targetLanguage: targetLanguage.value
    });
    
    if (response.success) {
      isRecording.value = true;
    } else {
      error.value = response.error;
      status.value = 'error';
    }
  } catch (err) {
    error.value = err.message;
    status.value = 'error';
  }
};

const stopRecording = async () => {
  try {
    status.value = 'processing';
    
    const response = await window.ipcRenderer.invoke('asr:stop');
    
    if (response.success) {
      isRecording.value = false;
      status.value = 'idle';
    } else {
      error.value = response.error;
      status.value = 'error';
    }
  } catch (err) {
    error.value = err.message;
    status.value = 'error';
  }
};

const resetResults = () => {
  transcriptionText.value = '';
  translationText.value = '';
  isFinalTranscription.value = false;
  isFinalTranslation.value = false;
  transcriptionDetails.value = null;
  translationDetails.value = null;
};

const calculateConfidence = (transcription: any): number => {
  if (!transcription?.words?.length) return 0;
  const fixedWords = transcription.words.filter((w: any) => w.fixed).length;
  return Math.round((fixedWords / transcription.words.length) * 100);
};

// 监听IPC事件
const setupIPCListeners = () => {
  const { ipcRenderer } = window;
  
  ipcRenderer.on('asr:result', (_, result: IPCChannels['asr:result']) => {
    if (result.transcription !== undefined) {
      transcriptionText.value = result.transcription;
      isFinalTranscription.value = result.isFinal;
    }
    if (result.translation !== undefined) {
      translationText.value = result.translation;
      isFinalTranslation.value = result.isFinal;
    }
  });
  
  ipcRenderer.on('asr:transcription', (_, data: IPCChannels['asr:transcription']) => {
    transcriptionDetails.value = data;
  });
  
  ipcRenderer.on('asr:translation', (_, data: IPCChannels['asr:translation']) => {
    translationDetails.value = data;
  });
  
  ipcRenderer.on('asr:status', (_, data: IPCChannels['asr:status']) => {
    status.value = data.status;
    if (data.error) {
      error.value = data.error;
    }
  });
  
  ipcRenderer.on('asr:error', (_, errorMsg: string) => {
    error.value = errorMsg;
    status.value = 'error';
  });
};

// 清理IPC监听器
const cleanupIPCListeners = () => {
  const { ipcRenderer } = window;
  ipcRenderer.removeAllListeners('asr:result');
  ipcRenderer.removeAllListeners('asr:transcription');
  ipcRenderer.removeAllListeners('asr:translation');
  ipcRenderer.removeAllListeners('asr:status');
  ipcRenderer.removeAllListeners('asr:error');
};

// 生命周期
onMounted(() => {
  setupIPCListeners();
});

onUnmounted(() => {
  cleanupIPCListeners();
  if (isRecording.value) {
    stopRecording();
  }
});
</script>

4.5 主进程集成 (src/main/main.ts)
import { app, BrowserWindow, ipcMain } from 'electron';
import { ASRIPC } from './ipc/asrBridge';
import { config } from './env';

// 确保环境变量已加载
console.log('API Key配置:', config.apiKey ? '已设置' : '未设置');

let mainWindow: BrowserWindow | null = null;
let asrIPC: ASRIPC | null = null;

async function createWindow() {
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    webPreferences: {
      nodeIntegration: false,
      contextIsolation: true,
      preload: path.join(__dirname, 'preload.js')
    }
  });

  // 加载Vue应用
  if (process.env.NODE_ENV === 'development') {
    await mainWindow.loadURL('http://localhost:5173');
  } else {
    await mainWindow.loadFile('dist/index.html');
  }

  // 初始化ASR IPC桥接
  asrIPC = new ASRIPC(mainWindow);

  // 开发工具
  if (process.env.NODE_ENV === 'development') {
    mainWindow.webContents.openDevTools();
  }
}

// 应用生命周期
app.whenReady().then(createWindow);

app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

app.on('activate', () => {
  if (BrowserWindow.getAllWindows().length === 0) {
    createWindow();
  }
});

app.on('before-quit', () => {
  // 清理资源
  if (asrIPC) {
    asrIPC.destroy();
  }
});
4.6 预加载脚本 (src/main/preload.js)
const { contextBridge, ipcRenderer } = require('electron');

// 安全地暴露IPC方法给渲染进程
contextBridge.exposeInMainWorld('ipcRenderer', {
  invoke: (channel, ...args) => ipcRenderer.invoke(channel, ...args),
  on: (channel, listener) => {
    const subscription = (event, ...args) => listener(...args);
    ipcRenderer.on(channel, subscription);
    return () => ipcRenderer.removeListener(channel, subscription);
  },
  removeAllListeners: (channel) => ipcRenderer.removeAllListeners(channel)
});
